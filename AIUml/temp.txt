@startuml
entity Company {
    * ID : int
    - name : varchar(40)
    - address : varchar(100)
    - phone : varchar(20)
}

entity Department {
    * ID : int
    - name : varchar(40)
    - manager : varchar(40)
    - create_date : date
    - company_id : int
}

entity Employee {
    * ID : int
    - name : varchar(40)
    - gender : varchar(10)
    - age : int
    - phone : varchar(20)
    - email : varchar(40)
    - department_id : int
}

entity Position {
    * ID : int
    - name : varchar(40)
    - salary : decimal
}

Department -|> Employee
Department -|> Position
Employee - Position
@enduml
